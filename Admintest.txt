-- HD Admin Delta - Minimal Universal Fix (No GUI - Chat Based)
-- Execute this in Delta Executor first

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local Prefix = "."

local function getPlayer(name)
    name = name:lower()
    for _, plr in pairs(Players:GetPlayers()) do
        if plr.Name:lower():find(name) or name == "all" or name == "others" or name == "me" then
            return plr
        end
    end
    return LocalPlayer
end

local Commands = {
    ["kill"] = function(args) local t = getPlayer(args[1] or "me") if t.Character and t.Character:FindFirstChild("Humanoid") then t.Character.Humanoid.Health = 0 end end,
    ["god"] = function(args) local t = getPlayer(args[1] or "me") if t.Character and t.Character:FindFirstChild("Humanoid") then t.Character.Humanoid.MaxHealth = 9e9 t.Character.Humanoid.Health = 9e9 end end,
    ["ff"] = function(args) local t = getPlayer(args[1] or "me") if t.Character then Instance.new("ForceField", t.Character) end end,
    ["unff"] = function(args) local t = getPlayer(args[1] or "me") if t.Character then for _,v in pairs(t.Character:GetChildren()) do if v:IsA("ForceField") then v:Destroy() end end end end,
    ["tp"] = function(args) local t1 = getPlayer(args[1] or "me") local t2 = getPlayer(args[2] or "me") if t1.Character and t2.Character and t1.Character:FindFirstChild("HumanoidRootPart") then t1.Character.HumanoidRootPart.CFrame = t2.Character.HumanoidRootPart.CFrame + Vector3.new(0,5,0) end end,
    ["ban"] = function(args) local t = getPlayer(args[1]) if t and t ~= LocalPlayer then if t.Character then t.Character:BreakJoints() end game.Players:Chat("/kick "..t.Name.." Banned") end end,
    ["speed"] = function(args) local t = getPlayer(args[1] or "me") local s = tonumber(args[2]) or 100 if t.Character and t.Character:FindFirstChild("Humanoid") then t.Character.Humanoid.WalkSpeed = s end end,
    ["fly"] = function(args) local t = getPlayer(args[1] or "me") if t.Character and t.Character:FindFirstChild("HumanoidRootPart") then local bv = Instance.new("BodyVelocity") bv.MaxForce = Vector3.new(4000,4000,4000) bv.Velocity = t.Character.HumanoidRootPart.CFrame.LookVector * 50 bv.Parent = t.Character.HumanoidRootPart game:GetService("Debris"):AddItem(bv,999) end end,
    ["noclip"] = function(args) local t = getPlayer(args[1] or "me") if t.Character then for _,p in pairs(t.Character:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = false end end end end,
    ["fling"] = function(args) local t = getPlayer(args[1] or "me") if t.Character and t.Character:FindFirstChild("HumanoidRootPart") then t.Character.HumanoidRootPart.Velocity = Vector3.new(0,500,0) end end,
}

LocalPlayer.Chatted:Connect(function(msg)
    if msg:sub(1,1) == Prefix then
        local cmdStr = msg:sub(2)
        local args = cmdStr:split(" ")
        local cmd = table.remove(args,1):lower()
        if Commands[cmd] then
            Commands[cmd](args)
        end
    end
end)

print("HD Admin Minimal Universal Loaded - Prefix: .")
print("Use in chat: .kill me | .ban player | .god all | .speed 500 | .fly")
