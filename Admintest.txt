-- Nyx Delta Script for Pls Donate - Immortal Noob Bacon Chatting Bots
-- Paste directly into Delta Roblox Executor while in Pls Donate

local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local ChatService = game:GetService("Chat")

local BOT_COUNT = 25
local BOT_SPEED = 28
local BOT_HEALTH = 999999999

local botNames = {
    "NyxNoob", "BaconNoob", "PlsDonateNoob", "ShadowBacon", "VoidNoob",
    "NeoBacon", "PhantomNoob", "RageBacon", "ApexNoob", "NightBacon",
    "ChaosNoob", "FrostBacon", "ThunderNoob", "BloodBacon", "NyxBacon"
}

local chatLines = {
    "pls give me Robux, I am Noob",
    "Pls donate Robux I am Noob :(",
    "I am Noob pls give me Robux",
    "Noob here, anyone give Robux pls",
    "Pls give me Robux I am Noob",
    "Help a Noob with Robux donate",
    "I am Noob pls Robux",
    "Donate Robux to Noob me",
    "Noob begging for Robux pls",
    "Give Robux I am Noob",
    "Pls Robux for Noob",
    "I am Noob need Robux donate",
    "Bacon Noob wants Robux pls",
    "Noob Bacon here donate"
}

local function CreateImmortalNoobBaconBot()
    local botModel = Instance.new("Model")
    local randomName = botNames[math.random(1, #botNames)] .. "_" .. math.random(100,999)
    botModel.Name = randomName
    
    local humanoid = Instance.new("Humanoid")
    humanoid.Health = BOT_HEALTH
    humanoid.MaxHealth = BOT_HEALTH
    humanoid.WalkSpeed = BOT_SPEED
    humanoid.JumpPower = 80
    humanoid.Parent = botModel
    
    -- Noob Bacon Body Parts
    local root = Instance.new("Part")
    root.Name = "HumanoidRootPart"
    root.Size = Vector3.new(2, 2, 1)
    root.Transparency = 0
    root.Color = Color3.fromRGB(0, 0, 0)
    root.Material = Enum.Material.Plastic
    root.CanCollide = false
    root.Parent = botModel
    
    local torso = Instance.new("Part")
    torso.Name = "Torso"
    torso.Size = Vector3.new(2, 2, 1)
    torso.Color = Color3.fromRGB(0, 0, 255)  -- Classic blue
    torso.Material = Enum.Material.Plastic
    torso.Parent = botModel
    
    local head = Instance.new("Part")
    head.Name = "Head"
    head.Size = Vector3.new(1.25, 1.25, 1.25)
    head.Color = Color3.fromRGB(255, 204, 153)  -- Noob skin
    head.Material = Enum.Material.SmoothPlastic
    head.Parent = botModel
    
    -- Bacon Hair
    local hair = Instance.new("Part")
    hair.Name = "Hair"
    hair.Size = Vector3.new(1.2, 1.2, 1.2)
    hair.Color = Color3.fromRGB(255, 102, 0)  -- Bacon color
    hair.Material = Enum.Material.Plastic
    hair.Parent = botModel
    
    local hairMesh = Instance.new("SpecialMesh")
    hairMesh.MeshType = Enum.MeshType.FileMesh
    hairMesh.MeshId = "rbxassetid://19358630"  -- Classic Bacon Hair
    hairMesh.Scale = Vector3.new(1.1, 1.1, 1.1)
    hairMesh.Parent = hair
    
    local leftArm = Instance.new("Part")
    leftArm.Name = "Left Arm"
    leftArm.Size = Vector3.new(1, 2, 1)
    leftArm.Color = Color3.fromRGB(255, 204, 153)
    leftArm.Parent = botModel
    
    local rightArm = Instance.new("Part")
    rightArm.Name = "Right Arm"
    rightArm.Size = Vector3.new(1, 2, 1)
    rightArm.Color = Color3.fromRGB(255, 204, 153)
    rightArm.Parent = botModel
    
    local leftLeg = Instance.new("Part")
    leftLeg.Name = "Left Leg"
    leftLeg.Size = Vector3.new(1, 2, 1)
    leftLeg.Color = Color3.fromRGB(0, 0, 255)
    leftLeg.Parent = botModel
    
    local rightLeg = Instance.new("Part")
    rightLeg.Name = "Right Leg"
    rightLeg.Size = Vector3.new(1, 2, 1)
    rightLeg.Color = Color3.fromRGB(0, 0, 255)
    rightLeg.Parent = botModel
    
    -- Welds for classic R6 Noob look
    local function weld(part0, part1, c0)
        local w = Instance.new("Motor6D")
        w.Part0 = part0
        w.Part1 = part1
        w.C0 = c0
        w.Parent = part0
    end
    
    weld(torso, head, CFrame.new(0, 1.5, 0))
    weld(torso, leftArm, CFrame.new(-1.5, 0.5, 0))
    weld(torso, rightArm, CFrame.new(1.5, 0.5, 0))
    weld(torso, leftLeg, CFrame.new(-0.5, -1.5, 0))
    weld(torso, rightLeg, CFrame.new(0.5, -1.5, 0))
    weld(head, hair, CFrame.new(0, 0.6, 0))
    weld(root, torso, CFrame.new(0, 0, 0))
    
    botModel.PrimaryPart = root
    botModel.Parent = Workspace
    
    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = head
    billboard.Size = UDim2.new(5, 0, 1.2, 0)
    billboard.StudsOffset = Vector3.new(0, 3.5, 0)
    billboard.AlwaysOnTop = true
    billboard.Parent = head
    
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 1, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = randomName .. " [NOOB BACON]"
    nameLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
    nameLabel.TextScaled = true
    nameLabel.Font = Enum.Font.Arcade
    nameLabel.Parent = billboard
    
    spawn(function()
        while botModel.Parent do
            wait(0.2)
            local closest = nil
            local minDist = math.huge
            for _, plr in pairs(Players:GetPlayers()) do
                if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                    local dist = (plr.Character.HumanoidRootPart.Position - root.Position).Magnitude
                    if dist < minDist and dist < 500 then
                        minDist = dist
                        closest = plr.Character.HumanoidRootPart
                    end
                end
            end
            if closest then
                humanoid:MoveTo(closest.Position)
            end
        end
    end)
    
    spawn(function()
        while botModel.Parent do
            wait(math.random(6, 14))
            local msg = chatLines[math.random(1, #chatLines)]
            ChatService:Chat(head, msg, Enum.ChatColor.Green)
        end
    end)
